[趣题] 将前520个素数分成差为521的两部分

· 2019-08-12 · # 背包问题 # 素数

大家都知道，前几天是七夕（其实已经好久了）。我在网上看到一道趣题：

证明：可以将前 520 个素数分成两部分，使得这两部分的差恰等于 521。

证明是不可能的，我们用010101背包解决这个问题。

计算质数并求和

我们先计算出前520520520个质数，并将他们求和。

#include<bits/stdc++.h>
using namespace std;
bool prime[10000];
int main(){
	freopen("2.txt","w",stdout);//质数表存储的地方
	for(int i=4;i<10000;i+=2)prime[i]=1;//先筛偶数
	for(int i=3;i<100;i+=2){//筛奇数
		for(int j=i<<1;j<10000;j+=i)prime[j]=1;
	}
	int cnt=0,i=2;
	while(cnt!=520){//只要前520个
		if(prime[i]==0){cnt++;cout<<i<<" ";}//输出并累加
		i++;//下标加一
	}
	return 0;
}

010101背包

我们用010101背包来计算这些质数相加都可以得到哪些数字。

证明

然后这个是本文中唯一的数学知识：算出这两组数的总和。

因为总和是897745897745897745，那么两组数的和一定是897745897745897745，列出方程a+b=897745a+b=897745a+b=897745。

然后他们的差是521521521。我们假使a>ba>ba>b，那么又列出方程a−b=521a-b=521a−b=521。

解出这个方程：a=449133,b=448612a=449133,b=448612a=449133,b=448612。

这个代码检验是否可以有质数相加等于448612448612448612：

#include<bits/stdc++.h>
using namespace std;
bool bag[1000000]={1};//记得0是一种放法
int a;
int main(){
	freopen("2.txt","r",stdin);//读入所有质数
	for(int i=0;i<520;i++){
		cin>>a;
		for(int j=999999;j>=0;j--){//从后面往前枚举
			if(bag[j]==1)bag[j+a]=1;//放
		}
	}
	cout<<bag[448612]<<endl;//检验
	return 0;
}

程序输出的是111，说明成立

#include<bits/stdc++.h>

	KKK://goto不要经常用
		for(int i=0;i<520;i++){
			for(int j=999999;j>=0;j--){
				if(bag[j]==1){
					bag[j+a[i]]=1;

						num=j;//记得要减去 等于num-=a[i]
						cout<<a[i]<<" ";//它就肯定是其中的一个

				}
			}
		}
		if(num==0)return 0;//如果分解完了结束
		memset(bag,0,sizeof(bag));//初始化
		bag[0]=1;//同上
		goto KKK;//返回
	return 0;
}

#include<bits/stdc++.h>
using namespace std;
int ans,a,ans2,ans3,aa,aa2,aa3;//ans总和 aa个数

int main(){
	ifstream I("2.txt");//读入质数
	while(I>>a){t[a]=1;ans3+=a;aa3++;}//质数置1
	ifstream I2("3.txt");//读入第二组数
	while(I2>>a){t[a]=0;ans2+=a;aa2++;}//质数置0
	freopen("4.txt","w",stdout);//另一组的路径
	for(int i=0;i<10000;i++)if(t[i]){cout<<i<<" ";ans+=i;aa++;}//如果为1说明没被选中，也就是另一组
	freopen("CON","w",stdout);//输出在控制台
	cout<<ans<<" "<<ans2<<" "<<ans3<<endl;//输出总和
	cout<<aa<<" "<<aa2<<" "<<aa3<<endl;//个数
}
